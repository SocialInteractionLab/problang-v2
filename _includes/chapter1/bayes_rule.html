import jax
import jax.numpy as jnp

# Bayes' rule example
def bayes_rule(prior, likelihood, evidence):
    """Calculate posterior using Bayes' rule"""
    return (likelihood * prior) / evidence

# Example: Medical test
prior_disease = 0.01  # 1% of population has disease
test_sensitivity = 0.95  # 95% true positive rate
test_specificity = 0.90  # 90% true negative rate

# Calculate evidence (total probability of positive test)
evidence = (test_sensitivity * prior_disease + 
           (1 - test_specificity) * (1 - prior_disease))

# Calculate posterior probability of disease given positive test
posterior = bayes_rule(prior_disease, test_sensitivity, evidence)

print(f"Prior probability of disease: {prior_disease:.1%}")
print(f"Test sensitivity: {test_sensitivity:.1%}")
print(f"Test specificity: {test_specificity:.1%}")
print(f"Evidence (P(positive test)): {evidence:.4f}")
print(f"Posterior probability of disease given positive test: {posterior:.1%}")
