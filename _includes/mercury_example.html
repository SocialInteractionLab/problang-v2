import mercury as mr
import memo
import numpy as np

# Mercury widget for array size
array_size = mr.Slider(value=5, min=3, max=10, label="Array Size", step=1)

# Create array based on widget value
X = np.arange(array_size.value)

@memo.memo
def simple_model[a: X, b: X]():
    return a + b

# Calculate the result
result = simple_model()

# Display results with Mercury
mr.Markdown(f"## Results for array size {array_size.value}")
mr.Markdown(f"**Result shape:** {result.shape}")
mr.Markdown("**Result values:**")
mr.DataFrame(result, label="Addition Matrix")
